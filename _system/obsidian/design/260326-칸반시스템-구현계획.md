- [ ] **Step 1: kanban_sync.py 생성**

  `_system/tools/kanban/kanban_sync.py`:
  ```python
  #!/usr/bin/env python3
  """
  칸반 보드 frontmatter sync 스크립트.
  칸반 보드 .md 파일을 파싱하여 연결된 파일의 frontmatter를 업데이트한다.

  사용법:
      uv run --project _system/tools python _system/tools/kanban/kanban_sync.py
  또는:
      _system/tools/kanban/kanban-sync.sh
  """

  import re
  import sys
  import yaml
  from pathlib import Path


  # ──────────────────────────────────────────
  # 1. 칸반 보드 파서
  # ──────────────────────────────────────────

  def parse_kanban_board(text: str) -> dict[str, list[str]]:
      """
      칸반 보드 MD 텍스트 파싱 → {column_name: [wikilink_targets]}

      - [[target|display]] → target 만 추출
      - - [x] 체크된 카드도 포함 (컬럼 위치가 frontmatter 진실)
      - '%% kanban:settings' 이후 라인은 파싱 중지
      """
      result: dict[str, list[str]] = {}
      current_column: str | None = None

      for line in text.splitlines():
          if line.startswith('%% kanban:settings'):
              break
          col_match = re.match(r'^## (.+)$', line)
          if col_match:
              current_column = col_match.group(1).strip()
              result[current_column] = []
              continue
          if current_column is not None:
              # - [ ] [[target]] 또는 - [x] [[target|display]]
              card_match = re.match(r'^- \[.\] \[\[([^\]|]+)(?:\|[^\]]+)?\]\]', line)
              if card_match:
                  result[current_column].append(card_match.group(1).strip())

      return result


  # ──────────────────────────────────────────
  # 2. 볼트 파일 탐색기
  # ──────────────────────────────────────────

  # 탐색 제외 폴더
  EXCLUDED_DIRS = {'.obsidian', '.git', '.claude', '_system'}


  def find_file_in_vault(vault_root: Path, wikilink_target: str) -> 'Path | None':
      """
      wikilink_target을 볼트 내에서 탐색하여 Path 반환.

      - 단순 파일명: "250926-drop-0001-huxe-ai" → 확장자 없이 일치하는 .md 검색
      - 경로 포함: "PRJ-볼트v3/README" → 상위 폴더까지 확인하여 더 정확한 매칭
      - 없으면 None, 중복이면 첫 번째 + 경고 출력
      - _system/, .git/ 등 시스템 폴더 제외
      """
      target_path = Path(wikilink_target)
      target_name = target_path.name          # "PRJ-볼트v3/README" → "README"
      has_parent = len(target_path.parts) > 1  # 경로 포함 여부
      expected_parent = target_path.parts[-2] if has_parent else None

      matches: list[Path] = []
      for md_file in vault_root.rglob('*.md'):
          if any(part in EXCLUDED_DIRS for part in md_file.relative_to(vault_root).parts):
              continue
          if md_file.stem != target_name:
              continue
          if has_parent and md_file.parent.name != expected_parent:
              continue
          matches.append(md_file)

      if not matches:
          return None
      if len(matches) > 1:
          print(f'  ⚠️  중복 파일 [[{wikilink_target}]]: {len(matches)}개 발견 → 첫 번째 사용: {matches[0].relative_to(vault_root)}')
      return matches[0]


  # ──────────────────────────────────────────
  # 3. Frontmatter 읽기/쓰기
  # ──────────────────────────────────────────

  def read_frontmatter(file_path: Path) -> dict:
      """MD 파일에서 frontmatter dict 반환. frontmatter 없으면 {}."""
      text = file_path.read_text(encoding='utf-8')
      if not text.startswith('---'):
          return {}
      end = text.find('\n---', 3)
      if end == -1:
          return {}
      try:
          return yaml.safe_load(text[3:end].strip()) or {}
      except yaml.YAMLError:
          return {}


  def set_frontmatter_field(file_path: Path, field: str, value: str) -> bool:
      """
      frontmatter의 특정 필드를 value로 업데이트.

      - 현재 값이 같으면 파일 미수정 + False 반환 (멱등성)
      - 필드가 없으면 frontmatter 마지막에 추가
      - frontmatter 자체가 없으면 파일 맨 앞에 추가
      - 변경 시 True 반환
      """
      current = read_frontmatter(file_path)
      if current.get(field) == value:
          return False

      text = file_path.read_text(encoding='utf-8')

      if not text.startswith('---'):
          file_path.write_text(f'---\n{field}: {value}\n---\n{text}', encoding='utf-8')
          return True

      end = text.find('\n---', 3)
      if end == -1:
          return False

      fm_text = text[3:end]
      body = text[end + 4:]  # '\n---' 이후 (body 포함)

      field_pattern = re.compile(rf'^{re.escape(field)}:.*$', re.MULTILINE)
      if field_pattern.search(fm_text):
          new_fm = field_pattern.sub(f'{field}: {value}', fm_text)
      else:
          new_fm = fm_text.rstrip() + f'\n{field}: {value}'

      file_path.write_text(f'---{new_fm}\n---{body}', encoding='utf-8')
      return True


  # ──────────────────────────────────────────
  # 4. 보드별 sync
  # ──────────────────────────────────────────

  def sync_board(board_path: Path, board_config: dict, vault_root: Path) -> tuple[int, int, int]:
      """
      보드 파일 파싱 → 각 카드의 frontmatter 업데이트.
      Returns: (changed, skipped, warned)
      """
      field = board_config['field']
      columns = board_config['columns']

      board_columns = parse_kanban_board(board_path.read_text(encoding='utf-8'))

      changed = skipped = warned = 0

      for col_name, targets in board_columns.items():
          fm_value = columns.get(col_name)
          if fm_value is None:
              print(f'  ⚠️  알 수 없는 컬럼: "{col_name}" — 건너뜀')
              warned += 1
              continue

          for target in targets:
              target_file = find_file_in_vault(vault_root, target)
              if target_file is None:
                  print(f'  ⚠️  파일 없음: [[{target}]] — 건너뜀')
                  warned += 1
                  continue

              updated = set_frontmatter_field(target_file, field, fm_value)
              if updated:
                  rel = target_file.relative_to(vault_root)
                  print(f'  ✅ {rel} → {field}: {fm_value}')
                  changed += 1
              else:
                  skipped += 1

      return changed, skipped, warned


  # ──────────────────────────────────────────
  # 5. 메인
  # ──────────────────────────────────────────

  def main() -> None:
      script_path = Path(__file__).resolve()
      vault_root = script_path.parents[3]  # kanban/ → tools/ → _system/ → [볼트명]/

      config_path = script_path.parent / 'kanban-config.yaml'
      boards_dir = vault_root / '_system' / 'obsidian' / 'kanban'

      if not config_path.exists():
          print(f'❌ 설정 파일 없음: {config_path}')
          sys.exit(1)
      if not boards_dir.exists():
          print(f'❌ 보드 폴더 없음: {boards_dir}')
          sys.exit(1)

      with open(config_path, encoding='utf-8') as f:
          config = yaml.safe_load(f)

      boards_config: dict = config.get('boards', {})
      total_changed = total_skipped = total_warned = 0

      for board_name, board_config in boards_config.items():
          board_file = boards_dir / f'{board_name}.md'
          if not board_file.exists():
              print(f'⚠️  보드 파일 없음: {board_file.name} — 건너뜀')
              total_warned += 1
              continue

          print(f'\n📋 {board_name} 처리 중...')
          c, s, w = sync_board(board_file, board_config, vault_root)
          total_changed += c
          total_skipped += s
          total_warned += w

      print(f'\n{"─" * 40}')
      if total_changed == 0 and total_warned == 0:
          print('✅ 모두 최신 상태 — 변경 없음')
      else:
          print(f'완료: 업데이트 {total_changed}개 / 스킵 {total_skipped}개 / 경고 {total_warned}개')


  if __name__ == '__main__':
      main()
  ```

보드 파일과 frontmatter가 모두 준비된 상태에서 sync 스크립트를 실행해 전체 파이프라인을 검증한다.

- [ ] **Step 1: 현재 frontmatter 스냅샷 기록**

  ```bash
  grep -r "drop_status:\|session_status:\|project_status:" \
    [볼트-상위경로]/[볼트명]/04_studio/drops/ \
    [볼트-상위경로]/[볼트명]/02_sessions/ \
    [볼트-상위경로]/[볼트명]/03_projects/ \
    --include="*.md" | sort
  ```

- [ ] **Step 2: sync 스크립트 실행**

  ```bash
  cd [볼트-상위경로]/[볼트명]
  _system/tools/kanban/kanban-sync.sh
  ```

  Expected 출력 예시:
  ```
  🔄 칸반 sync 시작...
     볼트: [볼트-상위경로]/[볼트명]

  📋 칸반-드롭 처리 중...
  ────────────────────────────────────────
  ✅ 모두 최신 상태 — 변경 없음
  ```
  (이미 frontmatter를 직접 설정했으므로 대부분 스킵 예상)

- [ ] **Step 3: 멱등성 검증**

  동일 명령 재실행 → 동일 결과 확인 (파일 수정 없음)

  ```bash
  _system/tools/kanban/kanban-sync.sh
  ```
  Expected: `✅ 모두 최신 상태 — 변경 없음`

- [ ] **Step 4: 카드 이동 시뮬레이션**

  칸반-드롭.md에서 임시 테스트: `260326-drop-0016-테스트` 카드를 `📥 수집 중` 컬럼으로 수동 이동 (파일 직접 편집)

  ```

  ```
